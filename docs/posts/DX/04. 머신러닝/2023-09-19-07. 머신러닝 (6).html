1. PCA

  • 주성분 만들기

from sklearn.decomposition import PCA

n = x_train_s.shape[1]
for i in range(1,n+1):
    exec(f"pca{i} = PCA(n_components = {i})")
    exec(f"x_train_pca{i} = pca{i}.fit_transform(x_train_s)")
