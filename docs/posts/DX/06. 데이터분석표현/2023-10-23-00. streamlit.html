# streamlit 라이브러리와 datetime 모듈 불러오기
import streamlit as st
import pandas as pd
from datetime import datetime  

st.title('Unit 4. Input Widgets')
st.caption('참조사이트: https://docs.streamlit.io/library/api-reference/widgets')

## 버튼
st.header('1. Button')
if st.button("Say hello") :
    st.write("hello")
else :
    st.write("Good bye")

st.header('2. Radio button')

st.write("***")

## 라디오 버튼
genre = st.radio("좋아하는 영호 장르를 선택하세요",
                 ("코미디", "SF", "액션"))

if genre ==  "코미디" :
    st.write("유쾌하신 분이시군요")
if genre == "SF" :
    st.write("저도 SF 좋아합니다.")
else :
    st.write("멋지십니다.")

st.write("***")


## 체크박스
st.header('3. Checkbox')    # 😄

agree = st.checkbox("I agree")

if agree : 
    st.write("😄"*10)

e = st.checkbox("데이터를 조작하시겠습니까?")
titanic = pd.read_csv("https://raw.githubusercontent.com/huhshin/streamlit/master/data_titanic.csv")

if e :  
    st.write(titanic)

    
st.write("***")


st.header("4. select box")
## 셀렉트 박스
option = st.selectbox("어떻게 연락드릴까요?", ("Email", "Mobile phone","Office phone"))

st.write("네", option, "잘 알겟습니다.")

## 멀티 셀렉트
options = st.multiselect("좋아하는 색을 모두 선택하세요",
                         ["Greeen","Yellow", "Red", "Blue"],
                         ["Yellow","Red"])

st.write("선호 색상 : ")
for i in options :
    st.write(i)


st.markdown("***")  


## 텍스트, 문자, 숫자 입력 받기

st.header("5. 텍스트, 문자, 숫자 입력 받기")

title = st.text_input("최애 영화를 입력하세요", "Sound of Music")
st.write("당신이 가장 좋아하는 영화는 :", title)

number = st.number_input("Insert a number(1-10)", min_value = 1, max_value = 10, value = 5, step = 1)
st.write("The current number is", number)

ymd = st.date_input("When is your birthday",
                    datetime(2000, 9, 6))
st.write("Your birthday is : ", ymd)

st.markdown("***")
